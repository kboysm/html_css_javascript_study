클로저 : 내부함수가 외부함수의 맥락에 접글할 수 있는 것

예제1

function outter(){ // 외부함수
let title = "coding everybody";
    
            function inner(){ //내부함수

            alert(title);

            }
inner();
}

이너함수에서 title이라는 변수가 없으므로 아우터함수에 title을 찾아 씀 이너함수에서 외부함수의 변수를 가져다 씀 -> 클로저

예제2

function outter(){ // 외부함수
let title = "coding everybody";
    
            return function(){ //내부함수

            alert(title);

            }

        }
inner = outter();
inner();

inner라는 변수에 아우터 함수의 return function이 inner변수에 드간다 외부함수는 이미 죽었는데 내부함수에서 외부함수로 접근하여 내부함수만을 변수에 담아 사용
